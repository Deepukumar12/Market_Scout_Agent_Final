# --- Predictive Analysis Models ---

class PerformerMetric(BaseModel):
    competitor_id: str
    name: str
    change_velocity_score: int # 0-100, higher means more changes
    innovation_index: int # 0-100
    market_sentiment: str # Positive, Neutral, Negative
    predicted_trend: str # "Up", "Down", "Stable"
    trend_probability: float # 0.0 - 1.0

class PredictiveAnalysisResult(BaseModel):
    top_performers: List[PerformerMetric] # High change velocity
    stable_performers: List[PerformerMetric] # Low change velocity
    trending_predictions: List[PerformerMetric] # High probability "Up" trend
    analysis_timestamp: datetime

# --- Endpoints ---

@router.get("/predictive-pipeline", response_model=PredictiveAnalysisResult)
async def run_predictive_pipeline(
    user_id: str = "current_user" # In a real app, extract from token
):
    """
    Analyzes all competitors in the database to determine top performers, 
    stable entities, and predictive trends based on simulated 'change velocity'.
    """
    
    # 1. Fetch all competitors (Simulated DB fetch if db not fully populated, but let's try to fetch)
    # real_competitors = await db.client["scoutiq_db"]["competitors"].find().to_list(100)
    # For robust demo, we will simulate fetching "all added competitors" 
    # and if count is low, we generate some mock ones to make the chart look good.
    
    competitors_list = []
    try:
        if db.client:
            cursor = db.client["scoutiq_db"]["competitors"].find({})
            real_competitors = await cursor.to_list(length=100)
            for comp in real_competitors:
                competitors_list.append({
                    "id": str(comp["_id"]),
                    "name": comp["name"]
                })
    except Exception as e:
        print(f"DB Error: {e}")
        
    # If valid DB fetch is empty (fresh install), pad with mock data
    if len(competitors_list) < 5: 
        for i in range(5):
             competitors_list.append({
                 "id": str(uuid.uuid4()),
                 "name": f"{random.choice(COMPANY_PREFIXES)} {random.choice(COMPANY_SUFFIXES)}"
             })

    metrics = []

    for comp in competitors_list:
        # Simulate analysis of "Changes"
        # In a real engine, this would query the 'reports' collection and count 'features' added in last 30 days.
        change_score = random.randint(10, 99)
        innovation = random.randint(change_score - 10, change_score + 5)
        if innovation > 100: innovation = 100
        
        # Determine Trend
        trend = "Stable"
        if change_score > 75: trend = "Up"
        elif change_score < 30: trend = "Down"
        
        # Market Sentiment simulation
        sentiments = ["Positive", "Neutral", "Negative"]
        weights = [0.5, 0.3, 0.2] if trend == "Up" else [0.2, 0.4, 0.4]
        sentiment = random.choices(sentiments, weights=weights, k=1)[0]

        metrics.append(PerformerMetric(
            competitor_id=comp["id"],
            name=comp["name"],
            change_velocity_score=change_score,
            innovation_index=innovation,
            market_sentiment=sentiment,
            predicted_trend=trend,
            trend_probability=round(random.uniform(0.6, 0.95), 2)
        ))

    # Sort logic
    metrics.sort(key=lambda x: x.change_velocity_score, reverse=True)
    
    top_performers = metrics[:3] # Higest changes
    stable_performers = sorted(metrics, key=lambda x: x.change_velocity_score)[:3] # Lowest changes
    
    # Trending: High innovation + Positive sentiment
    trending = [m for m in metrics if m.predicted_trend == "Up" and m.market_sentiment == "Positive"]
    
    return PredictiveAnalysisResult(
        top_performers=top_performers,
        stable_performers=stable_performers,
        trending_predictions=trending[:3],
        analysis_timestamp=datetime.utcnow()
    )
